// REPLACE WITH YOUR GOOGLE SHEET CSV LINK

const MASTER_CSV_URL = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vTp-yQVU_VNNOR5aXQ7m-Xv7BiAgYKjBfJuT6lIETy6ksjqR-Lj2WW4DNqYGCNX2rfCpUtemiS1shes/pub?gid=2070907343&single=true&output=csv';


let allMasterData = [];


// Rule 1: 21-Day Conditional Formatting

function getLeadTimeStyle(firstPostStr, releaseDateStr) {

    if (!firstPostStr || !releaseDateStr || releaseDateStr.includes("Undecided") || releaseDateStr.includes("TBD")) return "";

    const first = new Date(firstPostStr);

    const release = new Date(releaseDateStr);

    if (isNaN(first) || isNaN(release)) return "";


    const diff = (release - first) / (1000 * 60 * 60 * 24);

    if (diff > 21) return "lead-green";

    if (diff < 10) return "lead-red";

    return ""; 

}


// Rules 2 & 3: Color Scaling

function getHeatColor(val, min, max, isBenchmarkCol = false) {

    let value = parseFloat(val.toString().replace(/[^0-9.]/g, '')) || 0;

    

    // Static Benchmarks for Columns D and I (Post Counts)

    if (isBenchmarkCol) {

        if (value >= 50) return `rgb(29, 185, 84)`;

        if (value <= 10) return `rgb(255, 68, 68)`;

        const ratio = (value - 10) / (50 - 10);

        let r = ratio < 0.5 ? 255 : Math.floor(255 - (ratio - 0.5) * 2 * 255);

        let g = ratio > 0.5 ? 255 : Math.floor(ratio * 2 * 255);

        return `rgb(${r}, ${g}, 100)`;

    }


    // Dynamic scale for the specific artist

    const ratio = (value - min) / (max - min || 1);

    let r = ratio < 0.5 ? 255 : Math.floor(255 - (ratio - 0.5) * 2 * 255);

    let g = ratio > 0.5 ? 255 : Math.floor(ratio * 2 * 255);

    return `rgb(${r}, ${g}, 100)`;

}


function renderArtistData(artistName) {

    const search = document.getElementById('global-search').value.toLowerCase();

    const artistData = allMasterData.filter(row => row[0].trim().toLowerCase() === artistName.toLowerCase());

    

    // Helper to get stats for the Artist-Specific color scale

    const getStats = (idx) => {

        const vals = artistData.map(r => parseFloat(r[idx].toString().replace(/[^0-9.]/g, '')) || 0);

        return { min: Math.min(...vals), max: Math.max(...vals) };

    };


    // Index mapping for "Core Priority Tracks" CSV structure

    // F=6 (1st Wk), I=9 (Avg Weekly), K=11 (Avg Artist Views), L=32 (ER), M=36 (UGC), N=37 (UGC Views)

    const statsF = getStats(6), statsI = getStats(9), statsK = getStats(11), 

          statsL = getStats(32), statsM = getStats(36), statsN = getStats(37);


    let html = `<table><thead><tr>

        <th>Song</th><th>1st Post</th><th>Pre-Posts (D)</th><th>Release Date</th>

        <th>1st Wk Streams</th><th>Total Streams</th><th>Avg Wk Streams</th><th>Total Posts (I)</th>

        <th>Avg Artist Views</th><th>Engagement %</th><th>Total UGC</th><th>UGC Views</th>

    </tr></thead><tbody>`;


    artistData.forEach(row => {

        if (!row[1].toLowerCase().includes(search)) return;


        html += `<tr>

            <td><strong>${row[1]}</strong></td>

            <td class="${getLeadTimeStyle(row[2], row[3])}">${row[2] || '-'}</td>

            <td class="heat-cell" style="background:${getHeatColor(row[12], 0, 0, true)}">${row[12] || '0'}</td>

            <td>${row[3] || '-'}</td>

            <td class="heat-cell" style="background:${getHeatColor(row[6], statsF.min, statsF.max)}">${row[6] || '-'}</td>

            <td>${row[7] || '-'}</td>

            <td class="heat-cell" style="background:${getHeatColor(row[9], statsI.min, statsI.max)}">${row[9] || '-'}</td>

            <td class="heat-cell" style="background:${getHeatColor(row[13], 0, 0, true)}">${row[13] || '0'}</td>

            <td class="heat-cell" style="background:${getHeatColor(row[11], statsK.min, statsK.max)}">${row[11] || '-'}</td>

            <td class="heat-cell" style="background:${getHeatColor(row[32], statsL.min, statsL.max)}">${row[32] || '-'}</td>

            <td class="heat-cell" style="background:${getHeatColor(row[36], statsM.min, statsM.max)}">${row[36] || '-'}</td>

            <td class="heat-cell" style="background:${getHeatColor(row[37], statsN.min, statsN.max)}">${row[37] || '-'}</td>

        </tr>`;

    });


    document.getElementById('table-output').innerHTML = html + `</tbody></table>`;

}


function init() {

    Papa.parse(MASTER_CSV_URL, {

        download: true,

        header: false,

        complete: (results) => {

            allMasterData = results.data.filter(r => r[0] && r[0] !== "ARTIST");

            

            // Build sidebar, including Clever

            const artists = [...new Set(allMasterData.map(r => r[0]))].sort();

            const list = document.getElementById('artist-list');

            

            artists.forEach(name => {

                const li = document.createElement('li');

                li.innerText = name;

                li.onclick = () => {

                    document.querySelectorAll('li').forEach(el => el.classList.remove('active'));

                    li.classList.add('active');

                    document.getElementById('current-artist').innerText = name;

                    document.getElementById('update-timestamp').innerText = `Last Updated: ${new Date().toLocaleTimeString()}`;

                    renderArtistData(name);

                };

                list.appendChild(li);

            });

        }

    });

}

init();
